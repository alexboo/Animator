Animator - jQuery Plugin

Description

Example code:

    $(document).ready(function(){

	    var i = 0;

	    $('.menu-box').each(function(){

		    $(this).Animator().addAnimation({'opacity': '0.6'}, 200, {delay: i * 150});

		    i = i+1;
	    });

	    $('.menu-box').mouseover(function(){

		    $(this).Animator().addAnimation({'opacity': '1', 'width': '400px', 'background-color': '#008800'}, 300);
	    });

	    $('.menu-box').mouseout(function(){
		    $(this).Animator().addAnimation({'opacity': '0.6', 'width': '200px', 'background-color': '#1AA0E1'}, 500);
	    });

	    $('#example-code-open').click(function(){
		    $('#example-code-block').toggle();
	    });

    });
